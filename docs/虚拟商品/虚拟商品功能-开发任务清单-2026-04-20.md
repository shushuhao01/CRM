  // 1. physical商品：正常返回

        WHEN p.virtual_delivery_type='resource_link' THEN 
          (SELECT COUNT(*) FROM resource_inventory WHERE product_id=p.id AND status='unused' AND tenant_id=?)

```typescript
// POST /api/orders/:id/audit
router.post('/:id/audit', authenticateToken, async (req, res) => {
  const { action, remark } = req.body; // action: 'approve' | 'reject'
  const orderId = req.params.id;

  const order = await db.query('SELECT * FROM orders WHERE id=?', [orderId]);
  if (!order.length) return res.status(404).json({ message: '订单不存在' });

  if (action === 'approve') {
    const orderProductType = order[0].order_product_type;
    const markType = order[0].mark_type;

    if (markType === 'virtual') {
      // 查询该订单所有虚拟商品的发货方式
      const virtualItems = await db.query(
        `SELECT p.virtual_delivery_type FROM order_items oi
         JOIN products p ON oi.product_id = p.id
         WHERE oi.order_id = ? AND p.product_type = 'virtual'`, [orderId]
      );

      const allNone = virtualItems.every(v => v.virtual_delivery_type === 'none');

      if (allNone) {
        // 无需发货型：审核通过直接完成
        await db.query(
          `UPDATE orders SET status='completed', completion_source='audit_auto_complete' WHERE id=?`,
          [orderId]
        );
      } else {
        // 卡密/资源型：进入虚拟待发货
        await db.query(
          `UPDATE orders SET status='virtual_pending' WHERE id=?`,
          [orderId]
        );
      }
    } else {
      // 普通/混合订单：进入普通待发货
      await db.query(
        `UPDATE orders SET status='pending_shipment' WHERE id=?`,
        [orderId]
      );
    }
  } else if (action === 'reject') {
    await db.query(`UPDATE orders SET status='audit_rejected' WHERE id=?`, [orderId]);
    // 释放预占的卡密/资源
    await db.query(
      `UPDATE card_key_inventory SET status='unused', reserved_order_id=NULL
       WHERE reserved_order_id=? AND status='reserved'`, [orderId]
    );
    await db.query(
      `UPDATE resource_inventory SET status='unused', reserved_order_id=NULL
       WHERE reserved_order_id=? AND status='reserved'`, [orderId]
    );
  }
});
```

**验证**：
- 虚拟订单(无需发货) 审核通过 -> status = completed, completion_source = audit_auto_complete
- 虚拟订单(卡密/资源) 审核通过 -> status = virtual_pending
- 普通订单 审核通过 -> status = pending_shipment
- 审核拒绝 -> 释放预占库存

import { Router } from 'express';
import { v4 as uuidv4 } from 'uuid';

const router = Router();

// POST /api/public/virtual-claim/login  客户登录
router.post('/login', async (req, res) => {
  const { token, phone, password, smsCode } = req.body;

  // 1. 校验令牌有效性
  const record = await db.query(
    `SELECT cki.*, o.customer_phone, vcs.login_methods, vcs.initial_password,
            vcs.claim_link_expiry_days
     FROM card_key_inventory cki
     JOIN orders o ON cki.order_id = o.id
     JOIN virtual_claim_settings vcs ON vcs.tenant_id = cki.tenant_id
     WHERE cki.claim_token = ? AND cki.status IN ('used','claimed')
     LIMIT 1`,
    [token]
  );
  // 也查resource_inventory，取并集
  if (!record.length) {
    return res.status(404).json({ message: '链接无效或已过期' });
  }

  // 2. 校验有效期
  const deliveryDate = record[0].updated_at;
  const expiryDays = record[0].claim_link_expiry_days || 30;
  const now = new Date();
  const diff = (now - new Date(deliveryDate)) / (1000 * 60 * 60 * 24);
  if (diff > expiryDays) {
    return res.status(410).json({ message: '领取链接已过期' });
  }

  // 3. 校验登录方式
  const loginMethods = record[0].login_methods; // 'sms' | 'password' | 'sms,password'
  if (loginMethods.includes('sms') && smsCode) {
    // 验证短信验证码（复用现有SMS服务）
    const valid = await verifySmsCode(phone, smsCode);
    if (!valid) return res.status(401).json({ message: '验证码错误' });
  } else if (loginMethods.includes('password') && password) {
    // 验证密码
    if (password !== record[0].initial_password) {
      return res.status(401).json({ message: '密码错误' });
    }
  } else {
    return res.status(400).json({ message: '请提供有效的登录凭证' });
  }

  // 4. 校验手机号匹配
  if (phone !== record[0].customer_phone) {
    return res.status(403).json({ message: '手机号与订单不匹配' });
  }

  // 5. 生成会话token
  const sessionToken = uuidv4();
  // 存入Redis或内存，有效期30分钟
  await cacheSet(`claim_session:${sessionToken}`, {
    claimToken: token,
    tenantId: record[0].tenant_id,
    phone
  }, 30 * 60);

  res.json({ sessionToken, message: '登录成功' });
});

// 获取邮箱配置（官方或自定义）
export async function getEmailConfig(tenantId: string): Promise<EmailConfig | null> {
  // 1. 查询租户虚拟商品领取配置
  const settings = await db.query(
    'SELECT email_source FROM virtual_claim_settings WHERE tenant_id=?', [tenantId]
  );
  const source = settings[0]?.email_source || 'official';

  if (source === 'custom') {
    // 查询租户自定义邮箱
    const config = await db.query(
      'SELECT * FROM tenant_email_config WHERE tenant_id=? AND is_verified=1', [tenantId]
    );
    if (!config.length) return null;
    return {
      host: config[0].smtp_host,
      port: config[0].smtp_port,
      encryption: config[0].encryption,
      email: config[0].sender_email,
      password: decrypt(config[0].sender_password), // 解密
      senderName: config[0].sender_name
    };
  } else {
    // 使用官方邮箱（管理后台配置的CRM邮箱）
    const officialConfig = await db.query(
      'SELECT * FROM admin_email_config WHERE id=1'
    );
    if (!officialConfig.length) return null;
    return {
      host: officialConfig[0].smtp_host,
      port: officialConfig[0].smtp_port,
      encryption: officialConfig[0].encryption,
      email: officialConfig[0].sender_email,
      password: decrypt(officialConfig[0].sender_password),
      senderName: officialConfig[0].sender_name || 'CRM系统'
    };
  }
}

// 发送虚拟商品邮件
export async function sendVirtualDeliveryEmail(params: {
  tenantId: string;
  toEmail: string;
  customerName: string;
  orderNo: string;
  items: Array<{ productName: string; type: string; content: string; instructions?: string }>;
  claimLink: string;
  contentMode: string; // 'link' | 'content' | 'both'
  systemName: string;
  systemAvatar: string;
}) {
  const config = await getEmailConfig(params.tenantId);
  if (!config) throw new Error('邮箱未配置');

  const transporter = nodemailer.createTransport({
    host: config.host,
    port: config.port,
    secure: config.encryption === 'ssl',
    auth: { user: config.email, pass: config.password }
  });

  // 生成HTML邮件内容
  const htmlContent = generateEmailTemplate(params);

  await transporter.sendMail({
    from: `"${config.senderName}" <${config.email}>`,
    to: params.toEmail,
    subject: `【${params.systemName}】您的虚拟商品已发货 - 订单${params.orderNo}`,
    html: htmlContent
  });
}
